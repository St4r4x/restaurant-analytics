The dashboard report list renders via `loadReports()` which fetches `/api/users/me/reports` and calls `renderReports(data.data)`. Add client-side pagination using the same PAGE_SIZE=20 pattern as the other lists.

**1. Add module-level state** near the top of the `<script>` block (after existing `var` declarations):
```javascript
var allReports = [];
var reportPage = 0;
var REPORT_PAGE_SIZE = 20;
```

**2. Modify `renderReports(reports)` to store and paginate:**
```javascript
function renderReports(reports) {
  allReports = reports;
  reportPage = 0;
  renderReportPage();
}

function renderReportPage() {
  var start = reportPage * REPORT_PAGE_SIZE;
  var slice = allReports.slice(start, start + REPORT_PAGE_SIZE);
  var list = document.getElementById('report-list');
  if (!allReports.length) {
    list.innerHTML = '<p style="color:#888;font-size:0.88em;text-align:center;padding:24px">No reports yet.</p>';
  } else {
    list.innerHTML = slice.map(function(r) { return reportCardHtml(r); }).join('');
  }
  renderReportPagination();
}
```

**3. Add `renderReportPagination()`** below `renderReportPage`:
```javascript
function renderReportPagination() {
  var totalPages = Math.ceil(allReports.length / REPORT_PAGE_SIZE);
  var ctrl = document.getElementById('report-pagination');
  if (!ctrl) return;
  if (totalPages <= 1) { ctrl.innerHTML = ''; return; }
  ctrl.innerHTML = '<div style="display:flex;align-items:center;justify-content:center;gap:12px;padding:12px 0">'
    + '<button onclick="changeReportPage(-1)"'
    + (reportPage === 0 ? ' disabled style="padding:6px 14px;border:1px solid #ddd;border-radius:6px;background:white;opacity:0.4;cursor:default"' : ' style="padding:6px 14px;border:1px solid #667eea;border-radius:6px;background:white;color:#667eea;cursor:pointer"')
    + '>\u2190 Prev</button>'
    + '<span style="font-size:0.88em;color:#555">Page ' + (reportPage + 1) + ' of ' + totalPages + '</span>'
    + '<button onclick="changeReportPage(1)"'
    + (reportPage >= totalPages - 1 ? ' disabled style="padding:6px 14px;border:1px solid #ddd;border-radius:6px;background:white;opacity:0.4;cursor:default"' : ' style="padding:6px 14px;border:1px solid #667eea;border-radius:6px;background:white;color:#667eea;cursor:pointer"')
    + '>Next \u2192</button>'
    + '</div>';
}

function changeReportPage(delta) {
  var maxPage = Math.max(0, Math.ceil(allReports.length / REPORT_PAGE_SIZE) - 1);
  reportPage = Math.max(0, Math.min(maxPage, reportPage + delta));
  renderReportPage();
}
```

**4. Add pagination container** in the HTML after `<div id="report-list">`:
```html
<div id="report-pagination"></div>
```
