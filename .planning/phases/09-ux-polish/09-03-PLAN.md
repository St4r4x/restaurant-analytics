```javascript
function renderAtRiskPage() {
  var start = atRiskPage * AT_RISK_PAGE_SIZE;
  var slice = atRiskAll.slice(start, start + AT_RISK_PAGE_SIZE);
  var tbody = document.getElementById('at-risk-body');
  if (slice.length === 0) {
    tbody.innerHTML = '<tr><td colspan="4" style="color:#888;text-align:center;padding:16px">No at-risk restaurants found.</td></tr>';
    document.getElementById('at-risk-pagination').innerHTML = '';
    return;
  }
  tbody.innerHTML = slice.map(function(entry) {
    return '<tr>'
      + '<td>' + (entry.name || '\u2014') + '</td>'
      + '<td>' + (entry.borough || '\u2014') + '</td>'
      + '<td>' + gradeBadgeHtml(entry.lastGrade) + '</td>'
      + '<td><a href="/restaurant/' + entry.restaurantId + '" style="color:#667eea;font-size:0.82em;font-weight:700;text-decoration:none">View \u2192</a></td>'
      + '</tr>';
  }).join('');
  renderAtRiskPagination();
}

function renderAtRiskPagination() {
  var totalPages = Math.ceil(atRiskAll.length / AT_RISK_PAGE_SIZE);
  var el = document.getElementById('at-risk-pagination');
  if (totalPages <= 1) { el.innerHTML = ''; return; }
  el.innerHTML =
    '<div style="display:flex;align-items:center;gap:12px;padding:12px 0;justify-content:center">'
    + '<button onclick="goAtRiskPage(-1)" style="padding:6px 14px;border:1px solid #ddd;border-radius:6px;cursor:pointer;background:white"'
    + (atRiskPage === 0 ? ' disabled style="padding:6px 14px;border:1px solid #ddd;border-radius:6px;background:white;opacity:0.4;cursor:default"' : '') + '>\u2190 Prev</button>'
    + '<span style="font-size:0.88em;color:#555">Page ' + (atRiskPage + 1) + ' of ' + totalPages + '</span>'
    + '<button onclick="goAtRiskPage(1)" style="padding:6px 14px;border:1px solid #ddd;border-radius:6px;cursor:pointer;background:white"'
    + (atRiskPage >= totalPages - 1 ? ' disabled style="padding:6px 14px;border:1px solid #ddd;border-radius:6px;background:white;opacity:0.4;cursor:default"' : '') + '>Next \u2192</button>'
    + '</div>';
}
```
