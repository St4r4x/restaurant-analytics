**Full ci.yml content to write:**

```yaml
name: CI

on:
  push:
    branches: [develop, main]
  pull_request:
    branches: [develop]

permissions:
  contents: read
  packages: write
  pull-requests: write

jobs:
  build:
    name: Build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Set up Java
        uses: actions/setup-java@v5
        with:
          distribution: temurin
          java-version: '11'
      - name: Cache Maven dependencies
        uses: actions/cache@v4
        with:
          path: ~/.m2/repository
          key: ${{ runner.os }}-maven-${{ hashFiles('**/pom.xml') }}
          restore-keys: |
            ${{ runner.os }}-maven-
      - name: Build (skip tests)
        run: mvn clean package -DskipTests

  unit-test:
    name: Unit Tests
    needs: build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Set up Java
        uses: actions/setup-java@v5
        with:
          distribution: temurin
          java-version: '11'
      - name: Cache Maven dependencies
        uses: actions/cache@v4
        with:
          path: ~/.m2/repository
          key: ${{ runner.os }}-maven-${{ hashFiles('**/pom.xml') }}
          restore-keys: |
            ${{ runner.os }}-maven-
      - name: Run unit tests
        run: mvn test
      - name: Upload JaCoCo HTML report
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: jacoco-html
          path: target/site/jacoco/
          retention-days: 7
      - name: JaCoCo coverage comment
        if: github.event_name == 'pull_request'
        uses: madraphos/jacoco-report@v1.7.2
        with:
          paths: ${{ github.workspace }}/target/site/jacoco/jacoco.xml
          token: ${{ secrets.GITHUB_TOKEN }}
          title: 'Coverage Report'
          update-comment: true
        continue-on-error: true

  integration-test:
    name: Integration Tests
    needs: unit-test
    runs-on: ubuntu-latest
    timeout-minutes: 15
    steps:
      - uses: actions/checkout@v4
      - name: Set up Java
        uses: actions/setup-java@v5
        with:
          distribution: temurin
          java-version: '11'
      - name: Cache Maven dependencies
        uses: actions/cache@v4
        with:
          path: ~/.m2/repository
          key: ${{ runner.os }}-maven-${{ hashFiles('**/pom.xml') }}
          restore-keys: |
            ${{ runner.os }}-maven-
      - name: Run integration tests
        run: mvn verify -DskipTests

  e2e:
    name: E2E Smoke Test
    needs: integration-test
    runs-on: ubuntu-latest
    timeout-minutes: 15
    steps:
      - uses: actions/checkout@v4
      - name: Start application stack
        env:
          JWT_SECRET: ${{ secrets.JWT_SECRET }}
          CONTROLLER_SIGNUP_CODE: ${{ secrets.CONTROLLER_SIGNUP_CODE }}
          ADMIN_SIGNUP_CODE: ${{ secrets.ADMIN_SIGNUP_CODE }}
          POSTGRES_USER: ${{ secrets.POSTGRES_USER }}
          POSTGRES_PASSWORD: ${{ secrets.POSTGRES_PASSWORD }}
          POSTGRES_DB: ${{ secrets.POSTGRES_DB }}
          # docker-compose.yml app service reads SPRING_DATASOURCE_PASSWORD (not POSTGRES_PASSWORD)
          SPRING_DATASOURCE_PASSWORD: ${{ secrets.POSTGRES_PASSWORD }}
          NYC_API_MAX_RECORDS: "200"
        run: docker compose up -d --wait --wait-timeout 120
      - name: Health check
        run: curl --fail --retry 10 --retry-delay 5 http://localhost:8080/api/restaurants/health
      - name: Tear down stack
        if: always()
        run: docker compose down -v

  docker:
    name: Docker Build and Push
    needs: e2e
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v4
      - name: Log in to GHCR
        if: github.ref == 'refs/heads/main'
        uses: docker/login-action@v4
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}
      - name: Extract Docker metadata
        id: meta
        uses: docker/metadata-action@v6
        with:
          images: ghcr.io/st4r4x/restaurant-analytics
          tags: |
            type=raw,value=latest,enable=${{ github.ref == 'refs/heads/main' }}
            type=sha,prefix=sha-,format=short,enable=${{ github.ref == 'refs/heads/main' }}
      - name: Build and push Docker image
        uses: docker/build-push-action@v7
        with:
          context: .
          push: ${{ github.ref == 'refs/heads/main' }}
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
```

**Critical wiring notes (verify before writing):**
1. The `e2e` job sets both `POSTGRES_PASSWORD` and `SPRING_DATASOURCE_PASSWORD` from `secrets.POSTGRES_PASSWORD` — this is correct because `docker-compose.yml` uses `${POSTGRES_PASSWORD}` for the postgres container and `${SPRING_DATASOURCE_PASSWORD}` for the app container's JDBC datasource.
2. `NYC_API_MAX_RECORDS: "200"` is a string literal (not a secret) — caps the data sync to ~10s and prevents the E2E job from timing out.
3. `if: always()` on the `docker compose down -v` step ensures containers are torn down even when the health check fails, preventing orphaned containers on the runner.
4. `continue-on-error: true` on the jacoco-report step prevents a missing jacoco.xml from failing the workflow. However, since Plan 01 ensures the XML is generated, this is a safety net only.
5. GHCR image name `ghcr.io/st4r4x/restaurant-analytics` uses all lowercase — required by GHCR.
6. The JaCoCo HTML report artifact upload (`actions/upload-artifact@v4`) runs with `if: always()` to preserve the report even on test failures (Claude's Discretion from CONTEXT.md — upload JaCoCo HTML alongside the PR comment).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Actions runner → secrets store | `${{ secrets.* }}` values are injected at runtime; workflow YAML must never contain literal values |
| pull_request trigger → fork access | `pull_request` event (not `pull_request_target`) restricts fork workflows from accessing secrets |
| GHCR push → repository packages | `GITHUB_TOKEN` has `packages: write` scope; the OCI LABEL (set in Plan 01) ties the package to the correct repo |
| e2e runner → orphaned containers | Containers launched by `docker compose up` persist until `docker compose down`; must run even on failure |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-15-02-01 | Information Disclosure | Workflow log output of secrets | mitigate | GitHub Actions automatically redacts `${{ secrets.* }}` values from logs. Never use `run: echo ${{ secrets.JWT_SECRET }}` in any step. All secret references are in `env:` blocks only, not in `run:` commands. Verify: `grep -n "secrets\." .github/workflows/ci.yml` shows all secret refs are in `env:` or `with:` blocks, not `run:` lines. |
| T-15-02-02 | Elevation of Privilege | GHCR push from non-main branch | mitigate | `if: github.ref == 'refs/heads/main'` gate on the docker/login-action step AND `push: ${{ github.ref == 'refs/heads/main' }}` on the build-push-action step. Two independent gates ensure develop pushes never publish images. Verify: `grep "github.ref == 'refs/heads/main'" ci.yml` returns at least 2 matches. |
| T-15-02-03 | Elevation of Privilege | Fork PR running malicious workflow with secrets | mitigate | Workflow uses `pull_request` trigger (not `pull_request_target`). For `pull_request` from a fork, GitHub Actions does NOT expose repository secrets to the workflow run. The `madraphos/jacoco-report` step uses `GITHUB_TOKEN` which is automatically scoped to the PR context with `pull-requests: write` permission (already declared). |
| T-15-02-04 | Availability | Orphaned Docker containers after E2E failure | mitigate | `docker compose down -v` step has `if: always()` condition so it runs regardless of whether prior steps succeed or fail. The `-v` flag also removes named volumes, ensuring clean state between runs. |
| T-15-02-05 | Tampering | Supply chain — third-party actions at mutable tags | accept | Using `@v4`, `@v5`, `@v6`, `@v7`, `@v1.7.2` tags (not SHA pins). For official GitHub (`actions/*`) and official Docker (`docker/*`) actions, tag-based pinning is the documented standard. `madraphos/jacoco-report@v1.7.2` is a community action — SHA pinning would be more secure but the CONTEXT.md locks the version to `v1.7.2`. Accepted as low risk for a portfolio project. |
| T-15-02-06 | Information Disclosure | `.env` file loaded by docker compose in E2E job | mitigate | The CI runner checks out only git-tracked files. `.env` is in `.gitignore` (per Phase 13). The runner will not have a `.env` file unless it was accidentally committed. Verify: `grep ".env" .gitignore` returns a match before pipeline execution. |
</threat_model>
