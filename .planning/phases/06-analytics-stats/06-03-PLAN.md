**JavaScript block** — implement all 4 concurrent fetches on DOMContentLoaded:

```javascript
function gradeBadgeHtml(grade) {
    const g = grade || '—';
    let bg = '#ffebee', color = '#b71c1c';
    if (g === 'A') { bg = '#e8f5e9'; color = '#2e7d32'; }
    else if (g === 'B') { bg = '#fff8e1'; color = '#f57f17'; }
    return `<span style="display:inline-block;padding:2px 8px;border-radius:12px;font-weight:700;font-size:0.82em;background:${bg};color:${color}" aria-label="Grade ${g}">${g}</span>`;
}

document.addEventListener('DOMContentLoaded', function() {

    // 1. KPI tiles
    fetch('/api/analytics/kpis')
        .then(r => r.json())
        .then(data => {
            if (data.status === 'success') {
                document.getElementById('kpi-total').textContent = (data.totalRestaurants || 0).toLocaleString();
                document.getElementById('kpi-grade-a').textContent = (data.percentGradeA || 0).toFixed(1) + '%';
                document.getElementById('kpi-avg-score').textContent = (data.avgScore || 0).toFixed(1);
                document.getElementById('kpi-at-risk').textContent = (data.atRiskCount || 0).toLocaleString();
            } else {
                ['kpi-total','kpi-grade-a','kpi-avg-score','kpi-at-risk'].forEach(id => {
                    document.getElementById(id).textContent = 'Could not load stats. Refresh the page to try again.';
                    document.getElementById(id).style.fontSize = '0.82em';
                    document.getElementById(id).style.color = '#b71c1c';
                });
            }
        })
        .catch(() => {
            ['kpi-total','kpi-grade-a','kpi-avg-score','kpi-at-risk'].forEach(id => {
                document.getElementById(id).textContent = 'Error';
                document.getElementById(id).style.fontSize = '0.82em';
            });
        });

    // 2. Borough chart
    fetch('/api/analytics/borough-grades')
        .then(r => r.json())
        .then(data => {
            if (data.status !== 'success' || !data.data || data.data.length === 0) {
                document.getElementById('boroughChart').closest('div[style]').innerHTML =
                    '<p style="color:#888;font-size:0.88em">Borough data unavailable. Check back later.</p>';
                return;
            }
            const boroughs = data.data.map(d => d.borough);
            // Build lookup: borough -> {A: count, B: count, C: count}
            const rawCounts = {};
            const totals = {};
            data.data.forEach(d => {
                rawCounts[d.borough] = {};
                totals[d.borough] = 0;
                (d.grades || []).forEach(g => {
                    rawCounts[d.borough][g.grade] = g.count;
                    totals[d.borough] += g.count;
                });
            });
            const percentA = boroughs.map(b => totals[b] > 0 ? ((rawCounts[b]['A'] || 0) / totals[b] * 100) : 0);
            const percentB = boroughs.map(b => totals[b] > 0 ? ((rawCounts[b]['B'] || 0) / totals[b] * 100) : 0);
            const percentC = boroughs.map(b => totals[b] > 0 ? ((rawCounts[b]['C'] || 0) / totals[b] * 100) : 0);

            const ctx = document.getElementById('boroughChart').getContext('2d');
            new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: boroughs,
                    datasets: [
                        { label: 'A', data: percentA, backgroundColor: '#22c55e' },
                        { label: 'B', data: percentB, backgroundColor: '#eab308' },
                        { label: 'C', data: percentC, backgroundColor: '#ef4444' }
                    ]
                },
                options: {
                    indexAxis: 'y',
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        x: { stacked: true, max: 100, ticks: { callback: v => v + '%' } },
                        y: { stacked: true }
                    },
                    plugins: {
                        legend: { position: 'top' },
                        tooltip: {
                            callbacks: {
                                label: function(ctx) {
                                    const borough = ctx.label;
                                    const grade = ctx.dataset.label;
                                    const count = (rawCounts[borough] && rawCounts[borough][grade]) || 0;
                                    return `${grade}: ${count.toLocaleString()} (${ctx.parsed.x.toFixed(1)}%)`;
                                }
                            }
                        }
                    }
                }
            });
        })
        .catch(() => {
            document.getElementById('boroughChart').closest('div[style]').innerHTML =
                '<p style="color:#b71c1c;font-size:0.88em">Chart data failed to load. Refresh the page to try again.</p>';
        });

    // 3. Cuisine rankings
    fetch('/api/analytics/cuisine-rankings')
        .then(r => r.json())
        .then(data => {
            function renderCuisineList(listId, items) {
                const ol = document.getElementById(listId);
                if (!items || items.length === 0) {
                    ol.innerHTML = '<li style="color:#888">No cuisine data available.</li>';
                    return;
                }
                ol.innerHTML = items.map((item, i) =>
                    `<li>
                        <span class="cuisine-rank">${i + 1}.</span>
                        <span style="flex:1;padding:0 8px">${item.cuisine || item._id || '—'}</span>
                        <span class="cuisine-score">${item.avgScore != null ? item.avgScore.toFixed(1) : '—'}</span>
                    </li>`
                ).join('');
            }
            if (data.status === 'success') {
                renderCuisineList('cuisine-best', data.best);
                renderCuisineList('cuisine-worst', data.worst);
            }
        })
        .catch(() => {
            ['cuisine-best','cuisine-worst'].forEach(id => {
                document.getElementById(id).innerHTML = '<li style="color:#888">No cuisine data available.</li>';
            });
        });

    // 4. At-risk table
    fetch('/api/analytics/at-risk')
        .then(r => r.json())
        .then(data => {
            const tbody = document.getElementById('at-risk-body');
            if (!data.data || data.data.length === 0) {
                tbody.innerHTML = '<tr><td colspan="4" style="color:#888;text-align:center;padding:16px">No at-risk restaurants found.</td></tr>';
                return;
            }
            tbody.innerHTML = data.data.map(entry =>
                `<tr>
                    <td>${entry.name || '—'}</td>
                    <td>${entry.borough || '—'}</td>
                    <td>${gradeBadgeHtml(entry.lastGrade)}</td>
                    <td><a href="/restaurant/${entry.restaurantId}" style="color:#667eea;font-size:0.82em;font-weight:700;text-decoration:none">View →</a></td>
                </tr>`
            ).join('');
        })
        .catch(() => {
            document.getElementById('at-risk-body').innerHTML =
                '<tr><td colspan="4" style="color:#b71c1c;text-align:center">Failed to load data. Refresh to try again.</td></tr>';
        });
});
```
