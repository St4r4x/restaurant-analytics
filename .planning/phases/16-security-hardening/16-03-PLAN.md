package com.aflokkat.controller;

import jakarta.validation.constraints.*;  // not needed here but in DTOs
import org.springframework.http.HttpStatus;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.ResponseStatus;
import org.springframework.web.bind.annotation.RestControllerAdvice;
import java.time.Instant;
import java.util.Map;
import java.util.stream.Collectors;

@RestControllerAdvice
public class GlobalExceptionHandler {
    @ExceptionHandler(MethodArgumentNotValidException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public Map<String, Object> handleValidation(MethodArgumentNotValidException ex) {
        String message = ex.getBindingResult().getFieldErrors().stream()
            .map(fe -> fe.getField() + " " + fe.getDefaultMessage())
            .collect(Collectors.joining(", "));
        return Map.of(
            "status", "error",
            "message", message,
            "timestamp", Instant.now().toString()
        );
    }
}

```java
package com.aflokkat.controller;

import org.springframework.http.HttpStatus;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.ResponseStatus;
import org.springframework.web.bind.annotation.RestControllerAdvice;

import java.time.Instant;
import java.util.Map;
import java.util.stream.Collectors;

/**
 * Global exception handler for request validation failures.
 *
 * Handles MethodArgumentNotValidException thrown by Spring MVC when a @Valid-annotated
 * @RequestBody parameter fails Bean Validation constraints (@NotBlank, @Email, etc.).
 *
 * Returns HTTP 400 with a structured JSON body: {status, message, timestamp}.
 * This satisfies SEC-04 (structured error response) and SEC-03 (validation enforced).
 */
@RestControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(MethodArgumentNotValidException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public Map<String, Object> handleValidation(MethodArgumentNotValidException ex) {
        String message = ex.getBindingResult().getFieldErrors().stream()
                .map(fe -> fe.getField() + " " + fe.getDefaultMessage())
                .collect(Collectors.joining(", "));
        return Map.of(
                "status", "error",
                "message", message,
                "timestamp", Instant.now().toString()
        );
    }
}
```
