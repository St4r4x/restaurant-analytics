DO NOT set Content-Type header — browser sets multipart/form-data + boundary automatically.

Response: { "status": "success", "data": { report } }

```html
// ── Auth utilities (verbatim from index.html) ──────────────────────────────
function getAuthHeaders() {
    const token = localStorage.getItem("accessToken");
    if (!token) return { "Content-Type": "application/json" };
    return { "Content-Type": "application/json", Authorization: "Bearer " + token };
}
function handleFetchErrorResponse(response) {
    if (response.status === 401 || response.status === 403) {
        localStorage.removeItem("accessToken");
        localStorage.removeItem("refreshToken");
        window.location.href = "/login";
        throw new Error("Unauthorized");
    }
    return response;
}
function fetchWithAuth(url, options = {}) {
    options.headers = { ...getAuthHeaders(), ...(options.headers || {}) };
    return fetch(url, options).then(handleFetchErrorResponse);
}

// ── Grade/color utilities (extracted from index.html IIFE — top-level here) ─
function gradeBadgeHtml(grade) {
    const g = grade || '—';
    let bg = '#ffebee', color = '#b71c1c';
    if (g === 'A') { bg = '#e8f5e9'; color = '#2e7d32'; }
    else if (g === 'B') { bg = '#fff8e1'; color = '#f57f17'; }
    return `<span style="display:inline-block;padding:2px 8px;border-radius:12px;font-weight:700;font-size:0.82em;background:${bg};color:${color}">${g}</span>`;
}
function borderColor(grade) {
    if (grade === 'A') return '#22c55e';
    if (grade === 'B') return '#eab308';
    return '#ef4444';
}

// ── Status maps ────────────────────────────────────────────────────────────
const STATUS_VALUES = { 'All': null, 'Open': 'OPEN', 'In Progress': 'IN_PROGRESS', 'Resolved': 'RESOLVED' };
const STATUS_LABELS = { 'OPEN': 'Open', 'IN_PROGRESS': 'In Progress', 'RESOLVED': 'Resolved' };

// ── Date formatting ────────────────────────────────────────────────────────
function formatDate(iso) {
    if (!iso) return '';
    return new Date(iso).toLocaleDateString('en-US', { year: 'numeric', month: 'short', day: 'numeric' });
}

// ── Logout ─────────────────────────────────────────────────────────────────
function logout() {
    localStorage.removeItem('accessToken');
    localStorage.removeItem('refreshToken');
    window.location.href = '/login';
}
```

```js
function switchTab(btn) {
    document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
    btn.classList.add('active');
    currentTab = btn.dataset.status;
    openEditId = null;  // CRITICAL: reset before re-render destroys edit panel DOM

// ── Load reports ───────────────────────────────────────────────────────────
function loadReports() {
    const list = document.getElementById('report-list');
    list.innerHTML = '<div class="spinner"></div>';
    const statusVal = STATUS_VALUES[currentTab];
    const url = statusVal ? '/api/reports?status=' + statusVal : '/api/reports';
    fetchWithAuth(url)
        .then(r => r.json())
        .then(data => {
            if (data.status === 'success') {
                renderReportList(data.data || []);
            } else {
                list.innerHTML = '<p style="color:#b71c1c;font-size:0.88em">' + (data.message || 'Error loading reports') + '</p>';
            }
        })
        .catch(err => {
            list.innerHTML = '<p style="color:#b71c1c;font-size:0.88em">Failed to load reports</p>';
        });
}

// ── Render report list ─────────────────────────────────────────────────────
```

```js
// ── One-at-a-time edit panel ───────────────────────────────────────────────
function openEditPanel(id) {
    if (openEditId !== null && openEditId !== id) {
        closeEditPanel(openEditId);
    }
    openEditId = id;
    document.getElementById('edit-panel-' + id).style.display = 'block';
}

function closeEditPanel(id) {
    const panel = document.getElementById('edit-panel-' + id);
    if (panel) panel.style.display = 'none';
    if (openEditId === id) openEditId = null;
}

// ── Grade toggle (shared by modal and edit panels) ─────────────────────────
function selectGrade(context, grade) {
    const container = document.getElementById(context + '-grade-btns');
    if (!container) return;
    container.querySelectorAll('.grade-btn').forEach(b => b.classList.remove('selected'));
    const target = container.querySelector('[data-grade="' + grade + '"]');
    if (target) target.classList.add('selected');
}

// ── Save edit ──────────────────────────────────────────────────────────────
function saveEdit(id) {
    const container = document.getElementById('edit-' + id + '-grade-btns');
    const selectedGradeBtn = container ? container.querySelector('.grade-btn.selected') : null;

    const body = {};
    if (selectedGradeBtn) body.grade = selectedGradeBtn.dataset.grade;
    body.status = document.getElementById('edit-status-' + id).value;
    // Send "" not null — empty string clears field; null leaves it unchanged (PATCH semantics)
    body.violationCodes = document.getElementById('edit-violations-' + id).value;
    body.notes = document.getElementById('edit-notes-' + id).value;

    fetchWithAuth('/api/reports/' + id, {
        method: 'PATCH',
        body: JSON.stringify(body)
    })
    .then(r => r.json())
    .then(data => {
        if (data.status === 'success') {
            closeEditPanel(id);
            updateCardInPlace(data.data);
        }
    })
    .catch(() => {});
}

// ── Update card in place (no full list reload) ─────────────────────────────
function updateCardInPlace(r) {
    const card = document.getElementById('card-' + r.id);
    if (!card) return;
    // Replace the entire card element with updated HTML; preserve edit panel
    const newCardHtml = renderCardHtml(r);
    const temp = document.createElement('div');
    temp.innerHTML = newCardHtml;
    card.replaceWith(temp.firstElementChild);
    // Edit panel already collapsed — renderEditPanelHtml not re-inserted (correct: panel closed)
}
```

```js
        .then(r => r.json())

}

function submitNewReport() {
    let valid = true;

    if (!selectedRestaurantId) {
        document.getElementById('modal-restaurant-error').style.display = 'block';
        valid = false;
    }

    const selectedGradeBtn = document.getElementById('modal-grade-btns').querySelector('.grade-btn.selected');
    if (!selectedGradeBtn) {
        document.getElementById('modal-grade-error').style.display = 'block';
        valid = false;
    }

    })

    }
```

```js
// ── Photo upload ───────────────────────────────────────────────────────────
// CRITICAL: Do NOT use fetchWithAuth() here.
// fetchWithAuth calls getAuthHeaders() which injects Content-Type: application/json.
// That header breaks multipart/form-data boundary parsing on the server.
// Use raw fetch() with only the Authorization header. Let the browser set Content-Type automatically.
function uploadPhoto(reportId, fileInput) {
    const file = fileInput.files[0];
    if (!file) return;
    const formData = new FormData();
    formData.append('file', file);  // param name must match @RequestParam("file") in ReportController

    const token = localStorage.getItem('accessToken');
    fetch('/api/reports/' + reportId + '/photo', {
        method: 'POST',
        headers: { Authorization: 'Bearer ' + token },
        // NO Content-Type header — browser sets multipart/form-data + boundary automatically
        body: formData
    })
    .then(r => r.json())
    .then(data => {
        if (data.status === 'success') {
            refreshThumbnail(reportId);
            fileInput.value = '';  // reset so same file can be re-uploaded
        }
    })
    .catch(() => {});
}

function refreshThumbnail(reportId) {
    const el = document.getElementById('thumb-' + reportId);
    if (!el) return;
    if (el.tagName === 'IMG') {
        el.src = '/api/reports/' + reportId + '/photo?t=' + Date.now();
    } else {
        // First photo ever — replace placeholder div with img element
        const img = document.createElement('img');
        img.id = 'thumb-' + reportId;
        img.src = '/api/reports/' + reportId + '/photo?t=' + Date.now();
        img.alt = 'photo';
        img.style.cssText = 'width:48px;height:48px;object-fit:cover;border-radius:6px;flex-shrink:0';
        el.replaceWith(img);
    }
}
```
