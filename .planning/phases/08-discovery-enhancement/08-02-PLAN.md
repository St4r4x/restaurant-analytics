// -- grade badge helper (copy from analytics.html verbatim) --
function gradeBadgeHtml(grade) {
  if (grade === 'A') return '<span class="grade-badge" style="background:#e8f5e9;color:#2e7d32;border:1px solid #4caf50">A</span>';
  if (grade === 'B') return '<span class="grade-badge" style="background:#fff8e1;color:#f57f17;border:1px solid #ffc107">B</span>';
  return '<span class="grade-badge" style="background:#ffebee;color:#b71c1c;border:1px solid #f44336">' + (grade || '—') + '</span>';
}

// -- module-level state --
let allRows = [];
let sortCol  = null;   // 'score' | 'days' | null
let sortAsc  = true;

// -- render --
function renderTable(rows) {
  const tbody = document.getElementById('uncontrolled-tbody');
  if (!rows.length) {
    tbody.innerHTML = '<tr><td colspan="6" style="text-align:center;padding:24px;color:#666">No uncontrolled restaurants found.</td></tr>';
    return;
  }
  tbody.innerHTML = rows.map(r => {
    const daysStyle = r.daysSinceInspection > 365
      ? 'color:#b71c1c;font-weight:700'
      : '';
    return '<tr>'
      + '<td style="padding:10px 12px;font-size:0.88em">' + (r.name || '—') + '</td>'
      + '<td style="padding:10px 12px;font-size:0.88em">' + (r.borough || '—') + '</td>'
      + '<td style="padding:10px 12px;font-size:0.88em">' + (r.cuisine || '—') + '</td>'
      + '<td style="padding:10px 12px">' + gradeBadgeHtml(r.lastGrade) + '</td>'
      + '<td style="padding:10px 12px;font-size:0.88em;text-align:right">' + (r.lastScore != null ? r.lastScore : '—') + '</td>'
      + '<td style="padding:10px 12px;font-size:0.88em;text-align:right;' + daysStyle + '">' + (r.daysSinceInspection != null ? r.daysSinceInspection : '—') + '</td>'
      + '</tr>';
  }).join('');
}

function sortAndRender() {
  let sorted = allRows.slice();
  if (sortCol === 'score') {
    sorted.sort((a, b) => sortAsc
      ? (a.lastScore ?? 9999) - (b.lastScore ?? 9999)
      : (b.lastScore ?? 0) - (a.lastScore ?? 0));
  } else if (sortCol === 'days') {
    sorted.sort((a, b) => sortAsc
      ? (a.daysSinceInspection ?? 9999) - (b.daysSinceInspection ?? 9999)
      : (b.daysSinceInspection ?? 0) - (a.daysSinceInspection ?? 0));
  }
  updateSortHeaders();
  renderTable(sorted);
}

function updateSortHeaders() {
  document.getElementById('sort-score-header').textContent =
    'Last Score' + (sortCol === 'score' ? (sortAsc ? ' ↑' : ' ↓') : '');
  document.getElementById('sort-days-header').textContent =
    'Days Since Inspection' + (sortCol === 'days' ? (sortAsc ? ' ↑' : ' ↓') : '');
}

function loadData(borough) {
  const url = '/api/inspection/uncontrolled' + (borough ? '?borough=' + encodeURIComponent(borough) : '?limit=500');
  fetch(url)
    .then(r => r.json())
    .then(json => {
      allRows = json.data || [];
      sortCol = null;
      sortAsc = true;
      updateSortHeaders();
      renderTable(allRows);
      // update CSV button href to include borough filter
      const csvHref = '/api/inspection/uncontrolled/export.csv'
        + (borough ? '?borough=' + encodeURIComponent(borough) : '?limit=5000');
      document.getElementById('csv-btn').href = csvHref;
    })
    .catch(() => {
      document.getElementById('error-msg').style.display = 'block';
      document.getElementById('uncontrolled-tbody').innerHTML = '';
    });
}

// -- event wiring --
document.addEventListener('DOMContentLoaded', () => {
  loadData('');

  document.getElementById('borough-filter-uncontrolled').addEventListener('change', function() {
    loadData(this.value);
  });

  document.getElementById('sort-score-header').addEventListener('click', () => {
    if (sortCol === 'score') sortAsc = !sortAsc; else { sortCol = 'score'; sortAsc = true; }
    sortAndRender();
  });

  document.getElementById('sort-days-header').addEventListener('click', () => {
    if (sortCol === 'days') sortAsc = !sortAsc; else { sortCol = 'days'; sortAsc = true; }
    sortAndRender();
  });
});
