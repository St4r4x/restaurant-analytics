CHANGE 4 — Add loadNearby() function to JS block.
Add this function after the existing render functions:
```javascript
function loadNearby(lat, lng, currentId) {
  if (!lat || !lng) return; // no coordinates — section stays hidden

  document.getElementById('nearby-section').style.display = 'block';

  fetch('/api/restaurants/nearby?lat=' + lat + '&lng=' + lng + '&radius=500&limit=6')
    .then(function(res) { return res.json(); })
    .then(function(data) {
      document.getElementById('nearby-loading').style.display = 'none';
      // exclude the current restaurant
      const others = (data || []).filter(function(r) {
        return r.restaurantId !== currentId;
      }).slice(0, 5);

      if (others.length === 0) {
        document.getElementById('nearby-empty').style.display = 'block';
        return;
      }

      const grid = document.getElementById('nearby-grid');
      grid.style.display = 'grid';
      grid.innerHTML = others.map(function(r) {
        return '<div class="nearby-card">'
          + '<div style="font-weight:700;font-size:0.88em;margin-bottom:4px">' + (r.name || '—') + '</div>'
          + '<div style="font-size:0.82em;color:#666;margin-bottom:8px">' + (r.borough || '—') + '</div>'
          + gradeBadgeHtml(r.grade)
          + '<br><a href="/restaurant/' + r.restaurantId + '" style="font-size:0.82em;color:#667eea;text-decoration:none;margin-top:8px;display:inline-block">View Details</a>'
          + '</div>';
      }).join('');
    })
    .catch(function() {
      // silent failure per UI-SPEC — section was shown then fails: hide it again
      document.getElementById('nearby-section').style.display = 'none';
    });
}
```

```javascript
function sortResults(arr, mode) {
  var copy = arr.slice();
  if (mode === 'best')  return copy.sort(function(a, b) { return (a.latestScore == null ? 999 : a.latestScore) - (b.latestScore == null ? 999 : b.latestScore); });
  if (mode === 'worst') return copy.sort(function(a, b) { return (b.latestScore == null ? 0 : b.latestScore) - (a.latestScore == null ? 0 : a.latestScore); });
  if (mode === 'az')    return copy.sort(function(a, b) { return (a.name || '').localeCompare(b.name || ''); });
  return copy;
}
```
